<script>


$(":checkbox").change(function() {
$('[id*=mrkt]').hide();

			$('input[type=checkbox]').each(function () {
											var show = this.value;

				if(this.checked) {
					localStorage.setItem(show, "true");
					if (show != "canafford") $('[id*='+show+']').show();
			
			} else localStorage.setItem(show, "false");



				
});
			
			if ($("#afford").is(':checked')) {
				$('[id*=notafford]').hide();
			}
			$('[id*=Teams]').hide();

});




var count = 0;

$( document ).ready( function() {
	$('[id*=mrkt]').hide();
	$('input[type=checkbox]').each(function () {
		var show = this.value;
		if(localStorage.getItem(show) == "true") {
			$(this).attr('checked', true);
			$('[id*='+show+']').show();
			$('[id*=Teams]').hide();
			
			count++;
		}
	});

if(count == 0) {
	localStorage.setItem("new", "true");
	localStorage.setItem("Potions", "true");
	$("input[value='new']").prop('checked', true);
	$("input[value='Potions']").prop('checked', true);
	$('[id*=new]').show();
	$('[id*=Potions]').show();
	$('[id*=Teams]').hide(); }
	
			if ($("#afford").is(':checked')) {
				$('[id*=notafford]').hide();
			}
			$('[id*=Teams]').hide();
});


</script>
